#define _FILE_OFFSET_BITS 64

#include <sys/types.h>
#include <dirent.h>
typedef struct dirent dirent_t;

const char* m3_readdir_name(DIR* dir)
{
    dirent_t* e;

    if (dir && (e = readdir(dir)))
        return e->d_name;

    return 0;
}
